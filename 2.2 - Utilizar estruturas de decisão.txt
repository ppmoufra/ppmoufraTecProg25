  viajante_rede_ip.c (versão com estruturas de decisão)

  Cada situação é controlada por estruturas de decisão (if/switch), simulando
  escolhas possíveis do viajante e desbravador de segredos.

  Compilar: gcc -o viajante_rede_ip viajante_rede_ip.c -std=c11
  Executar: ./viajante_rede_ip
*/

#include <stdio.h>

void situacao1() {
    int escolha;
    printf("(1) Você está na rede de IP. Três redes do pensamento são acessadas.\n");
    printf("Escolha: 1 - Acessar Rede A, 2 - Acessar Rede B, 3 - Acessar Rede C:\n");
    printf("Escolha: ");
    scanf("%d", &escolha);
    if (escolha == 1) printf("Você acessou a Rede A.\n");
    else if (escolha == 2) printf("Você acessou a Rede B.\n");
    else if (escolha == 3) printf("Você acessou a Rede C.\n");
    else printf("Rede desconhecida.\n");
}
