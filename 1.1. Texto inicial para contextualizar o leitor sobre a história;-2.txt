/*
  Desbravador de Segredos - Jogo interativo em C
  - Mínimo de 20 interações/decisões
  - Cada decisão tem 2+ opções (A, B, C...)
  - Modularizado: função ask_choice para reutilizar perguntas
  - Leitura de entrada segura usando fgets
*/

#include <stdio.h>
#include <string.h>
#include <ctype.h>

#define BUF_SIZE 128

/* Lê uma linha e retorna primeiro caractere válido (letra) em minúsculas.
   Retorna '\0' se entrada vazia. */
char read_choice_char() {
    char buf[BUF_SIZE];
    if (!fgets(buf, sizeof(buf), stdin)) return '\0';
    // encontrar primeiro caractere alfabético
    for (int i = 0; buf[i] != '\0'; ++i) {
        if (isalpha((unsigned char)buf[i])) {
            return (char)tolower((unsigned char)buf[i]);
        }
    }
    return '\0';
}

/* Pergunta reutilizável.
   - question: string com texto da pergunta
   - options: array de strings com opções (tamanho n_options)
   - n_options: número de opções (>=2)
   Retorna índice da opção escolhida (0..n_options-1)
*/
int ask_choice(const char *question, const char *options[], int n_options) {
    if (n_options < 2) return 0;
    while (1) {
        printf("\n%s\n", question);
        for (int i = 0; i < n_options; ++i) {
            printf("  %c) %s\n", 'A' + i, options[i]);
        }
        printf("Escolha uma opção (A-%c): ", 'A' + n_options - 1);
        char c = read_choice_char();
        if (c >= 'a' && c < 'a' + n_options) {
            int idx = (int)(c - 'a');
            printf(">> Você escolheu: %c) %s\n", 'A' + idx, options[idx]);
            return idx;
        } else {
            printf("Entrada inválida. Tente novamente.\n");
        }
    }
}

/* Mensagem breve de resultado imediato para cada escolha

            break;

            else printf("Você calibra manualmente — ajuste fino bem-sucedido.\n");
            break;

            else printf("Lógica matemática aplicada: precisão, pode perder intuição.\n");
            break;
        case 13:
            if (choice_index == 0) printf("Chave pública mostrada — autenticidade comprovada.\n");
            else printf("Demonstração em código: o guardião testa sua habilidade.\n");
            break;
        case 14:
            if (choice_index == 0) printf("Acesso temporário concedido — limites impostos.\n");
            else printf("Acesso permanente aceito — responsabilidade aumentada.\n");
            break;
        case 15:
            if (choice_index == 0) printf("Backup da tecnologia feito — material para estudo.\n");
            else printf("Deleta para proteger a origem — segurança priorizada.\n");
            break;

            if (choice_index == 0) printf("Aceita ver o futuro — imagens perturbadoras surgem.\n");

            if (choice_index == 0) printf("Rede Estável: rotina conhecida.\n");

            else printf("Reiniciar: nova rota será escolhida em breve.\n");

    printf("=== Desbravador de Segredos — Versão Console ===\n");
